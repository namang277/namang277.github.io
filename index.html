Delete at Last Node by Naman Gupta CSE-A 027

  1. START

  2. IF HEAD = NULL
    Write UNDERFLOW
    Go to Step 9
    [END OF IF]
    //If the first node is null or there is only one node, then they return null.
    3.  
  3. SET PTR = HEAD
    //ptr will point to head node
    4.  
  4. Repeat Steps 5 and 6 while PTR -> NEXT!= NULL
    //traversing two pointers, preptr to 2nd last node and ptr to last node
    5.  
  5. SET PREPTR = PTR
    6.  
  6. SET PTR = PTR -> NEXT
    END OF LOOP //exiting step 5 and 6
    7.  
  7. SET PREPTR -> NEXT = NULL
    //deleting the last node by pointing the second last node to NULL
    8.  
  8. FREE PTR
    //deallocating the memory of last node
    9.  
  9. EXIT
